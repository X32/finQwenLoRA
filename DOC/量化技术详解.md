## 1. 量化基础概念

### 1.1 什么是量化？

**定义**：量化是一种模型压缩技术，通过降低数值精度来减少模型大小和计算复杂度。

**数学表达**：
```
量化过程：f_quantize: ℝ → ℤ_q
反量化：f_dequantize: ℤ_q → ℝ

其中 ℝ 是实数集，ℤ_q 是有限整数集
```

**基本原理**：
```python
# 线性量化的数学表达
def quantize(x, scale, zero_point, q_min, q_max):
    """
    量化公式：x_q = round(x / scale) + zero_point
    反量化：x ≈ (x_q - zero_point) × scale
    """
    # 量化
    x_quantized = np.round(x / scale) + zero_point
    x_quantized = np.clip(x_quantized, q_min, q_max)
    
    # 反量化
    x_approx = (x_quantized - zero_point) * scale
    
    return x_quantized, x_approx

# 参数计算
scale = (x_max - x_min) / (q_max - q_min)
zero_point = -round(x_min / scale) - q_min
```

```
量化分类树：
量化

├─ 按量化时机
```

## 3. 整数量化 (INT4/INT8)

### 3.1 INT8 量化基础

**线性量化算法**：

```python
import numpy as np

class INT8Quantization:
    """INT8对称量化实现"""
    
    def __init__(self, symmetric=True, calibration_method="percentile"):
        self.symmetric = symmetric  # 对称量化
        self.calibration_method = calibration_method
        
    def calibrate(self, weights):
        """校准：确定量化范围"""
        if self.calibration_method == "minmax":
            # 简单的最小最大值
            w_min, w_max = weights.min(), weights.max()
            
        elif self.calibration_method == "percentile":
            # 百分位法，更鲁棒
            w_min = np.percentile(weights, 1)   # 排除异常值
            w_max = np.percentile(weights, 99)
            
        elif self.calibration_method == "mse":
            # 最小化量化误差
            w_min, w_max = self._optimize_range(weights)
            
        return w_min, w_max
    
    def quantize_symmetric(self, weights):
        """对称量化：范围关于0对称"""
        # 1. 确定量化范围
        w_max = max(abs(weights.min()), abs(weights.max()))
        
        # 2. 计算缩放因子
        # INT8范围：[-127, 127]（对称）
        q_max = 127
        scale = w_max / q_max
        
        # 3. 量化
        weights_int8 = np.round(weights / scale)
        weights_int8 = np.clip(weights_int8, -q_max, q_max)
        
        # 4. 反量化验证
        weights_approx = weights_int8 * scale
        
        # 5. 计算量化误差
        mse = np.mean((weights - weights_approx) ** 2)
        
        return weights_int8.astype(np.int8), scale, mse
    
    def quantize_asymmetric(self, weights):
        """非对称量化：可以更精确"""
        # 1. 确定范围
        w_min, w_max = weights.min(), weights.max()
        
        # 2. 计算缩放因子和零点
        # INT8范围：[-128, 127]（非对称）
        q_min, q_max = -128, 127
        scale = (w_max - w_min) / (q_max - q_min)
        zero_point = q_min - np.round(w_min / scale)
        
        # 3. 量化
        weights_int8 = np.round(weights / scale + zero_point)
        weights_int8 = np.clip(weights_int8, q_min, q_max)
        
        # 4. 反量化
        weights_approx = (weights_int8 - zero_point) * scale
        
        return weights_int8.astype(np.int8), scale, zero_point

# 使用示例
```

### 3.2 INT4 量化详解

**INT4 的挑战和解决方案**：

```python
class INT4Quantization:
    """INT4量化：极致压缩"""
    
    def __init__(self, group_size=128):
        """
        INT4 量化配置
        
        参数：
            group_size: 分块大小，128为经验最优值
        """
        self.group_size = group_size
        
    def blockwise_quantize(self, weights):
        """
        分块量化：解决INT4精度损失大的问题
        
        原理：将权重分成小块，每块独立量化
        """
        n = len(weights)
        quantized_blocks = []
        scales = []
        
        # 分块处理
        for i in range(0, n, self.group_size):
            block = weights[i:i+self.group_size]
            
            # 每块独立量化
            block_q, scale = self._quantize_block(block)
            quantized_blocks.append(block_q)
            scales.append(scale)
        
        # 合并结果
        weights_int4 = np.concatenate(quantized_blocks)
        
        return weights_int4, scales
    
    def _quantize_block(self, block):
        """单块量化"""
        # 对称量化：INT4范围 [-7, 7] 或 [-8, 7]
        block_max = max(abs(block.min()), abs(block.max()))
        q_max = 7  # 使用 [-7, 7] 范围
        
        # 计算该块的缩放因子
        scale = block_max / q_max if block_max > 0 else 1.0
        
        # 量化
        block_q = np.round(block / scale)
        block_q = np.clip(block_q, -q_max, q_max)
        
        return block_q.astype(np.int8), scale
```

### 3.3 GPTQ-INT4 技术

**GPTQ算法深度解析**：

```python
class GPTQQuantization:
    """
    GPTQ (Generative Pre-trained Transformer Quantization)
    专为Transformer设计的后训练4-bit量化
    """
    
    def __init__(self, bits=4, group_size=128, damp_percent=0.01):
        self.bits = bits
        self.group_size = group_size  
        self.damp_percent = damp_percent  # 阻尼系数
        
    def quantize_layer(self, layer_weight, hessian=None):
        """
        单层GPTQ量化
        
        关键创新：基于Hessian的感知量化
        """
        # 1. 如果没有Hessian，使用单位矩阵
        if hessian is None:
            hessian = np.eye(layer_weight.shape[1])
        
        # 2. 添加阻尼，提高数值稳定性
        hessian = hessian + np.diag(hessian.diagonal()) * self.damp_percent
        
        # 3. 分块处理
        blocks = self._split_into_blocks(layer_weight)
        quantized_blocks = []
        
        for i, block in enumerate(blocks):
            # 4. 提取对应的Hessian块
            hessian_block = hessian[i*self.group_size:(i+1)*self.group_size,
                                 i*self.group_size:(i+1)*self.group_size]
            
            # 5. GPTQ核心算法：OBQ (Optimal Brain Quantization)
            quantized_block = self._obq_quantize(block, hessian_block)
            quantized_blocks.append(quantized_block)
        
        # 6. 合并量化结果
        quantized_weight = np.concatenate(quantized_blocks)
        
        return quantized_weight
    
    def _obq_quantize(self, block, hessian):
        """
        最优脑量化：考虑权重间相互影响
        
        核心思想：量化一个权重时，补偿对其他权重的影响
        """
        n = len(block)
        quantized_block = block.copy()
        
        # 按重要性排序（Hessian对角线元素表示重要性）
        importance = np.diag(hessian)
        sorted_indices = np.argsort(-importance)  # 降序排列
        
        for idx, i in enumerate(sorted_indices):
            # 1. 量化当前权重
            scale = np.max(np.abs(block)) / (2**self.bits - 1)
            q_i = np.round(block[i] / scale)
            q_i = np.clip(q_i, -(2**(self.bits-1)), 2**(self.bits-1)-1)
            
            # 2. 计算量化误差
            delta = block[i] - q_i * scale
            
            # 3. 更新未量化的权重（关键步骤）
            for j in sorted_indices[idx+1:]:
                # 基于Hessian信息更新
                if hessian[i, i] != 0:
                    update = delta * hessian[i, j] / hessian[i, i]
                    block[j] += update
            
            quantized_block[i] = q_i
        
        return quantized_block
    
    def _split_into_blocks(self, weight):
        """将权重分块"""
        n = len(weight)
        blocks = []
        for i in range(0, n, self.group_size):
            block = weight[i:min(i+self.group_size, n)]
            blocks.append(block)
        return blocks
```

```python
class BNBINT4Quantization:
    """
    bitsandbytes库的INT4量化
    用于QLoRA训练和INT4推理
    """
    
    def __init__(self, quant_type="nf4", double_quant=True):
        """
        参数：
            quant_type: "nf4" (推荐) 或 "fp4"
            double_quant: 是否双重量化（推荐开启）
        """
        self.quant_type = quant_type
        self.double_quant = double_quant
        
    def nf4_quantize(self, weights):
        """
        NF4 (NormalFloat4) 量化
        
        核心思想：正态分布的分位数量化
        """
        # 1. 归一化权重到[-1, 1]
        weights_normalized = weights / np.max(np.abs(weights))
        
        # 2. 计算正态分布的分位数
        # NF4使用16个量化级别，基于标准正态分布的分位点
        quantiles = self._compute_nf4_quantiles()
        
        # 3. 量化：找到最近的量化级别
        weights_nf4 = self._map_to_quantiles(weights_normalized, quantiles)
        
        return weights_nf4
    
    def _compute_nf4_quantiles(self):
        """计算NF4的分位数表"""
        import scipy.stats as stats
        
        # 标准正态分布的16个分位点（INT4共16个值）
        # 对应[-7, -6, ..., 0, ..., 6, 7]的量化值
        quantiles = []
        for i in range(-8, 8):  # -7到7，加上两端
            if i == -8:
                q = -np.inf  # 负无穷
            elif i == 7:
                q = stats.norm.ppf((i + 0.5) / 16)  # 正半边的最后一个
            else:
                q = stats.norm.ppf((i + 8.5) / 16)  # 对称分布
            quantiles.append(q)
            
        return np.array(quantiles)
    
    def double_quantize(self, weights, scales):
        """
        双重量化：进一步压缩量化参数
        
        原理：对缩放因子再次量化
        """
        # 1. 标准量化
        weights_int4, scale = self._quantize_with_scale(weights)
        
        # 2. 对scale进行量化（通常8-bit）
        scale_q = np.round(scale / 0.01)  # 假设0.01精度
        scale_q = np.clip(scale_q, 0, 127).astype(np.uint8)
        
        # 3. 反量化
        scale_approx = scale_q * 0.01
        
        return weights_int4, scale_approx
```

```python
class W8A8Quantization:
    """
    Weight 8-bit + Activation 8-bit量化
    
    简称：W8A8
    效果：显著加速，精度损失小
    """
    
    def __init__(self, calibration_size=512):
        self.calibration_size = calibration_size
        
    def calibrate_activations(self, model, dataloader):
        """校准激活值范围"""
        # 收集激活值统计
        activation_stats = {}
        
        with torch.no_grad():
            for i, batch in enumerate(dataloader):
                if i >= self.calibration_size:
                    break
                    
                # 前向传播，记录激活值
                output = model(batch)
                
                # 收集每层的激活值范围
                for name, param in model.named_parameters():
                    if name not in activation_stats:
                        activation_stats[name] = []
                    activation_stats[name].append(output.abs().max())
        
        # 计算每层的量化范围
        quant_ranges = {}
        for name, stats in activation_stats.items():
            # 使用百分位数确定范围
            q_min = np.percentile(stats, 1)
            q_max = np.percentile(stats, 99)
            quant_ranges[name] = (q_min, q_max)
            
        return quant_ranges
    
    def quantize_model_w8a8(self, model, quant_ranges):
        """W8A8量化整个模型"""
        quantized_model = model.copy()
        
        for name, module in quantized_model.named_modules():
            if hasattr(module, 'weight'):
                # 权重量化：INT8
                weight = module.weight.data
                q_min, q_max = quant_ranges[name]
                
                # 对称量化权重
                scale = max(abs(q_min), abs(q_max)) / 127
                weight_int8 = torch.round(weight / scale).clamp(-127, 127)
                
                # 替换权重
                module.weight.data = weight_int8
                module.scale = scale
                
                # 激活值量化配置
                module.activation_range = quant_ranges[name]
        
        return quantized_model
```

```python
class W4A16Quantization:
    """
    Weight 4-bit + Activation 16-bit
    
    当前最流行的配置：
    - 权重：INT4 (极致压缩)
    - 激活值：FP16 (保持精度)
    """
    
    def __init__(self, group_size=128):
        self.group_size = group_size
        
    def quantize_weights_only(self, model):
        """
        只量化权重，不量化激活值
        
        优势：
        1. 训练友好（可在INT4权重上训练）
        2. 精度保持好（激活值保持FP16）
        3. 实现简单（只需量化权重）
        """
        for name, module in model.named_modules():
            if hasattr(module, 'weight'):
                weight = module.weight.data
                
                # 分块INT4量化
                weight_int4, scales = self._blockwise_int4(weight)
                
                # 存储量化权重和缩放因子
                module.weight.data = weight_int4
                module.scales = scales  # 每块一个scale
        
        return model
    
    def forward_with_dequantize(self, module, input):
        """
        前向传播：实时反量化
        
        流程：
        1. 输入：FP16激活值
        2. 权重：INT4 → FP16 反量化
        3. 计算：FP16矩阵乘法
        4. 输出：FP16激活值
        """
        # 1. 反量化权重
        weight_fp16 = self._dequantize_int4(module.weight, module.scales)
        
        # 2. 标准16位计算
        output = torch.nn.functional.linear(input, weight_fp16, module.bias)
        
        return output
    
    def _blockwise_int4(self, weight):
        """分块INT4量化"""
        n = len(weight)
        weight_int4 = []
        scales = []
        
        for i in range(0, n, self.group_size):
            block = weight[i:i+self.group_size]
            
            # 计算该块的缩放因子
            block_max = block.abs().max()
            scale = block_max / 7.0  # INT4范围 [-7, 7]
            
            # 量化
            block_q = torch.round(block / scale).clamp(-7, 7)
            weight_int4.append(block_q)
            scales.append(scale)
        
        return torch.cat(weight_int4), torch.tensor(scales)
```

### 7.1 决策树

**量化方法选择器**：

```python
def choose_quantization_method(requirements):
    """
    根据需求选择最佳量化方法
    
    参数：
        requirements: 包含task_type, hardware, precision_requirement等的字典
    """
    task_type = requirements.get('task_type')  # 'training' or 'inference'
    hardware = requirements.get('hardware')    # 'high-end', 'mid-range', 'edge'
    precision_req = requirements.get('precision')  # 'high', 'medium', 'low'
    memory_constraint = requirements.get('memory_constraint')
    
    # 决策逻辑
    if task_type == 'training':
        if memory_constraint == 'severe':
            return "BNB-4bit + LoRA", "极端显存受限"
        elif hardware == 'high-end':
            return "BF16", "大模型训练标准"
        else:
            return "FP16", "通用训练配置"
    
    elif task_type == 'inference':
        if hardware == 'edge':
            if precision_req == 'high':
                return "GPTQ-4bit", "边缘设备+精度要求"
            else:
                return "INT8", "边缘设备标准配置"
        
        elif hardware == 'high-end':
            if precision_req == 'high':
                return "FP16/BF16", "最高质量"
            elif precision_req == 'medium':
                return "INT8", "平衡速度和精度"
            else:
                return "GPTQ-4bit", "最大化吞吐量"
        
        else:  # mid-range
            if memory_constraint == 'tight':
                return "GPTQ-4bit", "显存受限"
            else:
                return "INT8", "通用部署"
    
    else:
        return "FP16", "默认安全选择"
```

## 💡 总结

### 核心要点

1. **FP32**: 训练基准，精度最高，显存占用大
2. **FP16**: 训练加速，显存减半，需要梯度处理
3. **BF16**: 大模型训练，数值范围大，稳定性好
4. **GPTQ-INT4**: 推理部署，极致压缩，精度保持好
5. **BNB-INT4**: 训练+推理，QLoRA基础，NF4量化

### 选择原则

- **训练优先**: FP16/BF16 → 必要时QLoRA
- **推理优先**: INT8平衡 → GPTQ极致
- **精度优先**: FP16/BF16 → INT8 → GPTQ
- **成本优先**: GPTQ → INT8 → FP16
